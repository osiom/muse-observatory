sudo apt update
sudo apt install postgresql postgresql-contrib

sudo -i -u postgres
psql

-- Create user with password
CREATE USER museuser WITH PASSWORD 'musepassword';

-- Create database owned by the user
CREATE DATABASE muse_observatory OWNER museuser;

-- Grant all privileges on the database to the user
GRANT ALL PRIVILEGES ON DATABASE muse_observatory TO museuser;

psql -U museuser -d muse_observatory -h localhost

sudo -i -u postgres psql -c "CREATE USER museuser WITH PASSWORD 'musepassword';"
sudo -i -u postgres psql -c "CREATE DATABASE muse_observatory OWNER museuser;"
sudo -i -u postgres psql -c "GRANT ALL PRIVILEGES ON DATABASE muse_observatory TO museuser;"
